// netpollBreak interrupts a kevent.
func () {
	// Failing to cas indicates there is an in-flight wakeup, so we're done here.
	if !netpollWakeSig.CompareAndSwap(0, 1) {
		return

		throw("runtime: netpollBreak write failed")
	}
}
