type lexer struct {
	input    string
	start    int
	pos      int
	line     int
	state    stateFn
	items    chan item
	tomlNext bool

	// Allow for backing up up to 4 runes. This is necessary because TOML
	// contains 3-rune tokens (""" and ''').
	prevWidths [4]int
	nprev      int  // how many of prevWidths are in use
	atEOF      bool // If we emit an eof, we can still back up, but it is not OK to call next again.

	// A stack of state functions used to maintain context.
	//
	// The idea is to reuse parts of the state machine in various places. For
	// example, values can appear at the top level or within arbitrarily nested
	// arrays. The last state on the stack is used after a value has been lexed.
	// Similarly for comments.
	stack []stateFn
}

type item struct {
	typ itemType
	val string
	err error
	pos Position
}

		panic("BUG in lexer: next called after EOF")

		.line++

// error stops all lexing by emitting an error and returning `nil`.
//
// Note that any value that is a character is escaped if it's a special
// character (newlines, tabs, etc.).

	if .atEOF {
		return .errorPrevLine()
	}
	.items <- item{typ: itemError, pos: .getPos(), err: }
	return nil

// lexTop consumes elements at the top level of TOML data.
func ( *lexer) stateFn {
	 := .next()
	if isWhitespace() || isNL() {
		return lexSkip(, )
	}
	switch  {
	case '#':
		.push()
		return lexCommentStart
	case '[':
		return lexTableStart
	case eof:
		if .pos > .start {
			return .errorf("unexpected EOF")
		}
		.emit(itemEOF)
		return nil
	}

	// At this point, the only valid item can be a key, so we back up
	// and let the key lexer do the rest.
	.backup()
	.push(lexTopEnd)
	return lexKeyStart
}

// lexTopEnd is entered whenever a top-level item has been consumed. (A value
// or a table.) It must see only whitespace, and will turn back to lexTop
// upon a newline. If it sees EOF, it will quit the lexer successfully.
func ( *lexer) stateFn {
	 := .next()
	switch {
	case  == '#':
		// a comment will read to a newline for us.
		.push(lexTop)
		return lexCommentStart
	case isWhitespace():
		return 
	case isNL():
		.ignore()
		return lexTop
	case  == eof:
		.emit(itemEOF)
		return nil
	}
	return .errorf(
		"expected a top-level item to end with a newline, comment, or EOF, but got %q instead",
		)
}

// lexTable lexes the beginning of a table. Namely, it makes sure that
// it starts with a character other than '.' and ']'.
// It assumes that '[' has already been consumed.
// It also handles the case that this is an item in an array of tables.
// e.g., '[[name]]'.
func ( *lexer) stateFn {
	if .peek() == '[' {
		.next()
		.emit(itemArrayTableStart)
		.push(lexArrayTableEnd)
	} else {
		.emit(itemTableStart)
		.push(lexTableEnd)
	}
	return lexTableNameStart
}

func ( *lexer) stateFn {
	.emit(itemTableEnd)
	return lexTopEnd
}

func ( *lexer) stateFn {
	.skip(isWhitespace)
	switch  := .peek(); {
	case  == ']' ||  == eof:
		return .errorf("unexpected end of table name (table names cannot be empty)")
	case  == '.':
		return .errorf("unexpected table separator (table names cannot be empty)")
	case  == '"' ||  == '\'':
		.ignore()
		.push(lexTableNameEnd)
		return lexQuotedName
	default:
		.push(lexTableNameEnd)
		return lexBareName
	}
}

// lexBareName lexes one part of a key or table.
//
// It assumes that at least one valid character for the table has already been
// read.
//
// Lexes only one part, e.g. only '"a"' inside '"a".b'.
func ( *lexer) stateFn {
	 := .next()
	switch {
	case isWhitespace():
		return lexSkip(, lexValue)

		.ignore() // ignore the '"'
		return lexString
	case  == '\'':
		.ignore() // ignore the "'"
		return lexRawString
	case  == eof:
		return .errorf("unexpected EOF; expected value")

// lexKeyStart consumes all key parts until a '='.
func ( *lexer) stateFn {
	.skip(isWhitespace)
	switch  := .peek(); {
	case  == '=' ||  == eof:
		return .errorf("unexpected '=': key name appears blank")
	case  == '.':
		return .errorf("unexpected '.': keys cannot start with a '.'")
	case  == '"' ||  == '\'':
		.ignore()
		fallthrough
	default: // Bare key
		.emit(itemKeyStart)
		return lexKeyNameStart
	}
}

func ( *lexer) stateFn {
	.skip(isWhitespace)
	switch  := .peek(); {

		return .errorf("unexpected '='")
	case  == '.':
		return .errorf("unexpected '.'")
	case  == '"' ||  == '\'':
		.ignore()
		.push(lexKeyEnd)
		return lexQuotedName
	default:
		.push(lexKeyEnd)
		return lexBareName

}

// lexKeyEnd consumes the end of a key and trims whitespace (up to the key
// separator).
func ( *lexer) stateFn {
	.skip(isWhitespace)
	switch  := .next(); {
	case isWhitespace():
		return lexSkip(, )
	case  == eof:
		return .errorf("unexpected EOF; expected key separator '='")
	case  == '.':
		.ignore()
		return lexKeyNameStart
	case  == '=':
		.emit(itemKeyEnd)
		return lexSkip(, lexValue)
	default:
		return .errorf("expected '.' or '=', but got %q instead", )
	}
}

// lexValue starts the consumption of a value anywhere a value is expected.
// lexValue will ignore whitespace.
// After a value is lexed, the last state on the next is popped and returned.
func ( *lexer) stateFn {
	// We allow whitespace to precede a value, but NOT newlines.
	// In array syntax, the array states are responsible for ignoring newlines.
	 := .next()
	switch {
	case isWhitespace():
		return lexSkip(, )
	case isDigit():
		.backup() // avoid an extra state and use the same as above
		return lexNumberOrDateStart
	}
	switch  {
	case '[':
		.ignore()
		.emit(itemArray)
		return lexArrayValue
	case '{':
		.ignore()
		.emit(itemInlineTableStart)
		return lexInlineTableValue
	case '"':
		if .accept('"') {
			if .accept('"') {
				.ignore() // Ignore """
				return lexMultilineString
			}
			.backup()
		}
		.ignore() // ignore the '"'
		return lexString
	case '\'':
		if .accept('\'') {
			if .accept('\'') {
				.ignore() // Ignore """
				return lexMultilineRawString
			}
			.backup()
		}
		.ignore() // ignore the "'"
		return lexRawString
	case '.': // special error case, be kind to users
		return .errorf("floats must start with a digit, not '.'")
	case 'i', 'n':
		if (.accept('n') && .accept('f')) || (.accept('a') && .accept('n')) {
			.emit(itemFloat)
			return .pop()
		}
	case '-', '+':
		return lexDecimalNumberStart
	}
	if unicode.IsLetter() {
		// Be permissive here; lexBool will give a nice error if the
		// user wrote something like
		//   x = foo
		// (i.e. not 'true' or 'false' but is something else word-like.)
		.backup()
		return lexBool
	}
	if  == eof {
		return .errorf("unexpected EOF; expected value")
	}
	return .errorf("expected value but found %q instead", )
}

// lexArrayValue consumes one value in an array. It assumes that '[' or ','
// have already been consumed. All whitespace and newlines are ignored.
func ( *lexer) stateFn {
	 := .next()
	switch {
	case isWhitespace() || isNL():
		return lexSkip(, )
	case  == '#':
		.push()
		return lexCommentStart
	case  == ',':
		return .errorf("unexpected comma")
	case  == ']':
		return lexArrayEnd
	}

	.backup()
	.push(lexArrayValueEnd)
	return lexValue
}

// lexArrayValueEnd consumes everything between the end of an array value and
// the next value (or the end of the array): it ignores whitespace and newlines
// and expects either a ',' or a ']'.
func ( *lexer) stateFn {
	switch  := .next(); {
	case isWhitespace() || isNL():
		return lexSkip(, )
	case  == '#':
		.push()
		return lexCommentStart
	case  == ',':
		.ignore()
		return lexArrayValue // move on to the next value
	case  == ']':
		return lexArrayEnd
	default:
		return .errorf("expected a comma (',') or array terminator (']'), but got %s", runeOrEOF())
	}
}

// lexArrayEnd finishes the lexing of an array.
// It assumes that a ']' has just been consumed.
func ( *lexer) stateFn {
	.ignore()
	.emit(itemArrayEnd)
	return .pop()
}

// lexInlineTableValue consumes one key/value pair in an inline table.
// It assumes that '{' or ',' have already been consumed. Whitespace is ignored.
func ( *lexer) stateFn {
	 := .next()
	switch {
	case isWhitespace():
		return lexSkip(, )
	case isNL():
		if .tomlNext {
			return lexSkip(, )
		}
		return .errorPrevLine(errLexInlineTableNL{})
	case  == '#':
		.push()
		return lexCommentStart
	case  == ',':
		return .errorf("unexpected comma")
	case  == '}':
		return lexInlineTableEnd
	}
	.backup()
	.push(lexInlineTableValueEnd)
	return lexKeyStart
}

// lexInlineTableValueEnd consumes everything between the end of an inline table
// key/value pair and the next pair (or the end of the table):
// it ignores whitespace and expects either a ',' or a '}'.
func ( *lexer) stateFn {
	switch  := .next(); {
	case isWhitespace():
		return lexSkip(, )
	case isNL():
		if .tomlNext {
			return lexSkip(, )
		}
		return .errorPrevLine(errLexInlineTableNL{})
	case  == '#':
		.push()
		return lexCommentStart
	case  == ',':
		.ignore()
		.skip(isWhitespace)
		if .peek() == '}' {
			if .tomlNext {
				return 
			}
			return .errorf("trailing comma not allowed in inline tables")
		}
		return lexInlineTableValue
	case  == '}':
		return lexInlineTableEnd
	default:
		return .errorf("expected a comma or an inline table terminator '}', but got %s instead", runeOrEOF())
	}
}

// lexString consumes the inner contents of a string. It assumes that the
// beginning '"' has already been consumed and ignored.
func ( *lexer) stateFn {
	 := .next()
	switch {
	case  == eof:
		return .errorf(`unexpected EOF; expected '"'`)
	case isNL():
		return .errorPrevLine(errLexStringNL{})
	case  == '\\':
		.push()
		return lexStringEscape
	case  == '"':
		.backup()
		.emit(itemString)
		.next()
		.ignore()
		return .pop()
	}
	return 
}

// lexMultilineString consumes the inner contents of a string. It assumes that
// the beginning '"""' has already been consumed and ignored.
func ( *lexer) stateFn {
	 := .next()
	switch  {
	default:
		return 
	case eof:
		return .errorf(`unexpected EOF; expected '"""'`)
	case '\\':
		return lexMultilineStringEscape
	case '"':
		/// Found "  try to read two more "".
		if .accept('"') {
			if .accept('"') {
				/// Peek ahead: the string can contain " and "", including at the
				/// end: """str"""""
				/// 6 or more at the end, however, is an error.
				if .peek() == '"' {
					/// Check if we already lexed 5 's; if so we have 6 now, and
					/// that's just too many man!
					///
					/// Second check is for the edge case:
					///
					///            two quotes allowed.
					///            vv
					///   """lol \""""""
					///          ^^  ^^^---- closing three
					///     escaped
					///
					/// But ugly, but it works
					if strings.HasSuffix(.current(), `"""""`) && !strings.HasSuffix(.current(), `\"""""`) {
						return .errorf(`unexpected '""""""'`)
					}
					.backup()
					.backup()
					return 
				}

				.backup() /// backup: don't include the """ in the item.
				.backup()
				.backup()
				.emit(itemMultilineString)
				.next() /// Read over ''' again and discard it.
				.next()
				.next()
				.ignore()
				return .pop()
			}
			.backup()
		}
		return 
	}
}

// lexRawString consumes a raw string. Nothing can be escaped in such a string.
// It assumes that the beginning "'" has already been consumed and ignored.
func ( *lexer) stateFn {
	 := .next()
	switch {
	default:
		return 
	case  == eof:
		return .errorf(`unexpected EOF; expected "'"`)
	case isNL():
		return .errorPrevLine(errLexStringNL{})
	case  == '\'':
		.backup()
		.emit(itemRawString)
		.next()
		.ignore()
		return .pop()
	}
}

// lexMultilineRawString consumes a raw string. Nothing can be escaped in such a
// string. It assumes that the beginning triple-' has already been consumed and
// ignored.
func ( *lexer) stateFn {
	 := .next()
	switch  {
	default:
		return 
	case eof:
		return .errorf(`unexpected EOF; expected "'''"`)
	case '\'':
		/// Found '  try to read two more ''.
		if .accept('\'') {
			if .accept('\'') {
				/// Peek ahead: the string can contain ' and '', including at the
				/// end: '''str'''''
				/// 6 or more at the end, however, is an error.
				if .peek() == '\'' {
					/// Check if we already lexed 5 's; if so we have 6 now, and
					/// that's just too many man!
					if strings.HasSuffix(.current(), "'''''") {
						return .errorf(`unexpected "''''''"`)
					}
					.backup()
					.backup()
					return 
				}

				.backup() /// backup: don't include the ''' in the item.
				.backup()
				.backup()
				.emit(itemRawMultilineString)
				.next() /// Read over ''' again and discard it.
				.next()
				.next()
				.ignore()
				return .pop()
			}
			.backup()
		}
		return 
	}
}

// lexMultilineStringEscape consumes an escaped character. It assumes that the
// preceding '\\' has already been consumed.
func ( *lexer) stateFn {
	if isNL(.next()) { /// \ escaping newline.
		return lexMultilineString
	}
	.backup()
	.push(lexMultilineString)
	return lexStringEscape()
}

func ( *lexer) stateFn {
	 := .next()
	switch  {
	case 'e':
		if !.tomlNext {
			return .error(errLexEscape{})
		}
		fallthrough
	case 'b':
		fallthrough
	case 't':
		fallthrough
	case 'n':
		fallthrough
	case 'f':
		fallthrough
	case 'r':
		fallthrough
	case '"':
		fallthrough
	case ' ', '\t':
		// Inside """ .. """ strings you can use \ to escape newlines, and any
		// amount of whitespace can be between the \ and \n.
		fallthrough
	case '\\':
		return .pop()
	case 'x':
		if !.tomlNext {
			return .error(errLexEscape{})
		}
		return lexHexEscape
	case 'u':
		return lexShortUnicodeEscape
	case 'U':
		return lexLongUnicodeEscape
	}
	return .error(errLexEscape{})
}
